# defining a function to read csv files for simplicity and avoiding repititive codes
def read_csv(fileName):
    return pd.read_csv(fileName , sep=',')
files = {}
names = ['AllstarFull.csv','Appearances.csv','AwardsPlayers.csv','Salaries.csv','Batting.csv' , 'Master.csv' , 'FieldingPost.csv', 'Pitching.csv']
for name in names:
    files[name.replace(".csv","")] = read_csv(name)
    
for key,val in files.items():
    exec(key + '=val')
